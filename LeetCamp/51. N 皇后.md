class Solution:
    def solveNQueens(self, n: int) -> List[List[str]]:
        def genetateBoard():
            board = list()
            for i in range(n):
                row[queens[i]] = "Q"
                board.append("".join(row))
                row[queens[i]] = "."
            return board

        def backtrack(row: int):
            if row == n:
                board = genetateBoard()
                solutions.append(board)
            else:
                for i in range(n):
                    if i in columns or row - i in diagnoal1 or row + i in diagnoal2:
                        continue
                    queens[row] = i
                    columns.add(i)
                    diagnoal1.add(row - i)
                    diagnoal2.add(row + i)
                    backtrack(row + 1)
                    columns.remove(i)
                    diagnoal1.remove(row - i)
                    diagnoal2.remove(row + i)

        solutions = list()
        queens = [-1] * n
        columns = set()
        diagnoal1 = set()
        diagnoal2 = set()
        row = ["."] * n
        backtrack(0)
        return solutions
