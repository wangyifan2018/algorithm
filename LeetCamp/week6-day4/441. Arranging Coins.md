class Solution:
    def arrangeCoins(self, n: int) -> int:
        left, right = 0, n
        while left  < right:
            mid = left + (right - left + 1) // 2
            if mid * (mid + 1) <= 2 * n:
                left = mid
            else:
                right = mid - 1
        return left
