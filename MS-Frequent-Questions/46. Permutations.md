class Solution:
    def permute(self, nums: List[int]) -> List[List[int]]:
        n = len(nums)
        res = []
        used = [False] * n
        self.dfs(nums, n, 0, used, [], res)
        return res

    def dfs(self, nums, n, depth, used, path, res):
        if depth == n:
            res.append(path.copy())
            return 
        for i in range(n):
            if not used[i]:
                path.append(nums[i])
                used[i] = True
                self.dfs(nums, n, depth + 1, used, path, res)
                path.pop()
                used[i] = False
