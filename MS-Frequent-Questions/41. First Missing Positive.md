class Solution:
    def firstMissingPositive(self, nums: List[int]) -> int:
        size = len(nums)
        for i in range(size):
            while 1 <= nums[i] <= size and nums[nums[i] - 1] != nums[i]:
                self.swap(nums, i, nums[i] - 1)
        
        for i in range(size):
            if nums[i] != i + 1:
                return i + 1
        return size + 1

    def swap(self, nums, idx1, idx2):
        nums[idx1], nums[idx2] = nums[idx2], nums[idx1]
