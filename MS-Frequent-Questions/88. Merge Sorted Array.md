class Solution:
    def merge(self, nums1: List[int], m: int, nums2: List[int], n: int) -> None:
        """
        Do not return anything, modify nums1 in-place instead.
        """
        tail = len(nums1) - 1
        m = m - 1
        n = n - 1
        while n >= 0 :
            while m >= 0 and nums1[m] > nums2[n]:
                nums1[m], nums1[tail] = nums1[tail], nums1[m]
                m -= 1
                tail -= 1
            nums1[tail] = nums2[n]
            n -= 1
            tail -= 1
        return nums1
