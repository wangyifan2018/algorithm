```python
# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def addTwoNumbers(self, l1: ListNode, l2: ListNode) -> ListNode:
        dummy = ListNode(0)
        cur = dummy
        flag = 0
        while l1 or l2:
            value1 = l1.val if l1 else 0
            value2 = l2.val if l2 else 0
            sum_v = value1 + value2 + flag
            flag = sum_v // 10
            cur.next = ListNode(sum_v % 10)
            cur = cur.next
            if l1: l1 = l1.next 
            if l2: l2 = l2.next 
        if flag:
            cur.next = ListNode(flag)
        return dummy.next
```
