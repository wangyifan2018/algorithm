class Solution:
    def reverseWords(self, s: str) -> str:
        left, right = 0, len(s) - 1
        while s[left] == ' ':
            left += 1
        while s[right] == ' ':
            right -= 1
        
        q = collections.deque()
        tmp = ''
        while left <= right:
            if s[left] == ' ' and len(tmp) > 0:
                q.appendleft(tmp)
                tmp = ''
            elif s[left] != ' ':
                tmp += s[left]
            left += 1
        q.appendleft(tmp)
        return ' '.join(q)
