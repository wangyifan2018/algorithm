```
# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def deleteDuplicates(self, head: ListNode) -> ListNode:
        if not head:
            return head
        dummy = ListNode(0)
        res = dummy
        dup = {}
        cur = head
        while cur:
            dup[cur.val] = dup.get(cur.val, 0) + 1
            cur = cur.next
        cur = head
        while cur:
            if dup[cur.val] > 1:
                cur = cur.next
                continue
            else:
                res.next = ListNode(cur.val)
                res = res.next
            cur = cur.next
        return dummy.next
```

一次遍历
```python
# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def deleteDuplicates(self, head: ListNode) -> ListNode:
        if not head:
            return head
        dummy = ListNode(0, head)
        cur = dummy
        while cur.next and cur.next.next:
            if cur.next.val == cur.next.next.val:
                x = cur.next.val
                while cur.next and cur.next.val == x:
                    cur.next = cur.next.next
            else:
                cur = cur.next
        return dummy.next
```
