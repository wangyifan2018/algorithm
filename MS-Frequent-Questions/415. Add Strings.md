```python
class Solution:
    def addStrings(self, num1: str, num2: str) -> str:
        l1 = len(num1) - 1
        l2 = len(num2) - 1
        res = ""
        up = 0
        while l1 >= 0 or l2 >= 0:
            n1 = num1[l1] if l1 >= 0 else 0
            n2 = num2[l2] if l2 >= 0 else 0
            tmp = int(n1) + int(n2) + up
            up = tmp // 10
            res = str(tmp % 10) + res
            l1 -= 1
            l2 -= 1
        return res if not up else "1" + res
```
