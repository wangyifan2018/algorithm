```python
# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def isPalindrome(self, head: ListNode) -> bool:
        val = []
        cur = head
        while cur:
            val.append(cur.val)
            cur = cur.next
        left, right = 0, len(val) - 1
        while left < right:
            if val[left] != val[right]:
                return False
            left += 1
            right -= 1
        return True
```
